# Use DataTable Skill

To heavily accelerate the development of Admin Dashboards and CRUD interfaces, the frontend implements a **Generic DataTable wrapper** powered by `@tanstack/react-table` and visually styled with `shadcn-ui`.

This component receives raw data arrays and rendering column structures, automatically generating a fully functional table with an interactive Tailwind CSS pagination footer.

## Component Location
The required components already exist in the frontend project:
- `<DataTable />`: `src/components/ui/data-table.tsx`
- `<DataTablePagination />`: `src/components/ui/data-table-pagination.tsx`

## Implementing a Table View

To build a table for any entity (e.g., Users, Trucks, Trips) generated by the Backend:

1. Import your respective API Hook generated by Orval.
2. Maintain local React state (`pageIndex`, `pageSize`) to drive the Query limits.
3. Pass `pageIndex` and `pageSize` to your Orval call. **Important:** Watch the types generated by Orval. Usually, NestJS pagination expects `{ page: pageIndex, pageSize: pageSize }`, **not** `limit`.
4. Define your `ColumnDef` rules.

### Example Implementation

```tsx
"use client"

import { useEffect, useState } from 'react';
import { ColumnDef } from '@tanstack/react-table';
import { usersControllerFindAll } from '@/src/api/generated/identity/identity/identity';
import { UserResponseDto } from '@/src/api/generated/identity/models';
import { DataTable } from '@/src/components/ui/data-table';

export default function GenericPage() {
  const [data, setData] = useState<UserResponseDto[]>([]);
  const [isLoading, setIsLoading] = useState(true);
  
  // DataTable uses 1-indexed pages and 10 rows by default
  const [pageCount, setPageCount] = useState(1);
  const [pageIndex, setPageIndex] = useState(1);
  const [pageSize, setPageSize] = useState(10);

  useEffect(() => {
    let active = true;
    async function fetchData() {
      setIsLoading(true);
      try {
        const response = await usersControllerFindAll({
          page: pageIndex,
          pageSize: pageSize, // Crucial: match the generated Orval param exactly
        });
        
        if (active) {
          // Flatten standard paginated response
          const items = Array.isArray(response) ? response : (response as any).items || [];
          const pages = (response as any).meta?.totalPages || 1;
          
          setData(items);
          setPageCount(pages);
        }
      } catch (error) {
        if (active) setData([]);
      } finally {
        if (active) setIsLoading(false);
      }
    }
    fetchData();
    return () => { active = false; };
  }, [pageIndex, pageSize]);

  // 1. Define how columns look and what object property they bind to
  const columns: ColumnDef<UserResponseDto>[] = [
    { accessorKey: 'id', header: 'ID' },
    { accessorKey: 'name', header: 'User Name' },
    { accessorKey: 'email', header: 'Email Address' },
  ];

  // 2. Render the generic wrapper
  return (
    <DataTable
      columns={columns}
      data={data}
      pageCount={pageCount}
      isLoading={isLoading}
      onPaginationChange={(newPage, newLimit) => {
        setPageIndex(newPage);
        setPageSize(newLimit);
      }}
    />
  );
}
```
